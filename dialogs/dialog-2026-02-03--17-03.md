  1. 按`F5`或点击工具栏的"本地Windows调试器"绿色箭头
  2. 程序将在断点处暂停
  3. 使用`F10`（单步跳过）、`F11`（单步进入）调试
  4. 在"自动"或"局部变量"窗口查看变量值

  **步骤5**：查看测试输出
  1. 调试过程中，打开`视图` → `输出`窗口
  2. 查看Google Test的测试结果

  /// Convert string to int, handling conversion errors gracefully.
  ///
  /// \param name the name of the variable to look up
  /// \param def  a default value to return if variable doesn't exist or 
  conversion fails
  ///
  /// \returns the integer value of environment variable \a name, or \a def if 
  the
  ///          variable is not set or cannot be converted
  ///
  inline int dotenv::getenv_int(const char* name, int def)
  {
      const char* str = std::getenv(name);
      if (!str) return def;

      std::string value(str);
      trim(value);

      if (value.empty()) {
          std::cout << "dotenv: Variable " << name << " is empty, using default 
  value" << std::endl;
          return def;
      }

      try {
          size_t pos;
          int result = std::stoi(value, &pos);
          // Check if entire string was consumed
          if (pos != value.length()) {
              std::cout << "dotenv: Variable " << name << " contains invalid 
  characters after number: '"
                       << value << "', using default value" << std::endl;
              return def;
          }
          return result;
      } catch (const std::exception& e) {
          std::cout << "dotenv: Failed to convert " << name << "='" << value
                   << "' to int: " << e.what() << ", using default value" <<
  std::endl;
          return def;
      }
  }

  ///
  /// Convert string to bool, supporting multiple formats.
  /// Recognizes: true/false, yes/no, on/off, 1/0 (case-insensitive).
  ///
  /// \param name the name of the variable to look up
  /// \param def  a default value to return if variable doesn't exist or 
  conversion fails
  ///
  /// \returns the boolean value of environment variable \a name, or \a def if 
  the
  ///          variable is not set or cannot be converted
  ///
  inline bool dotenv::getenv_bool(const char* name, bool def)
  {
      const char* str = std::getenv(name);
      if (!str) return def;

      std::string value(str);
      trim(value);

      // Convert to lowercase for case-insensitive comparison
      std::transform(value.begin(), value.end(), value.begin(),
                     [](unsigned char c){ return std::tolower(c); });

      if (value == "true" || value == "yes" || value == "on" || value == "1") {
          return true;
      } else if (value == "false" || value == "no" || value == "off" || value ==
   "0" || value.empty()) {
          return false;
      } else {
          std::cout << "dotenv: Cannot convert " << name << "='" << str
                   << "' to bool, using default value" << std::endl;
          return def;
      }
  }
